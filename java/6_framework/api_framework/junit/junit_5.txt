https://www.baeldung.com/junit-5-conditional-test-execution
	Conditional Test Execution with Annotations
		Operating System Conditions
			@EnabledOnOs({OS.WINDOWS, OS.MAC})
			@DisabledOnOs(OS.LINUX)
		Java Runtime Environment Conditions
			@EnabledOnJre({JRE.JAVA_10, JRE.JAVA_11})
			@EnabledForJreRange(min = JRE.JAVA_8, max = JRE.JAVA_13)
			@DisabledForJreRange(min = JRE.JAVA_14, max = JRE.JAVA_15)
			@DisabledOnJre(JRE.OTHER)
		System Property Conditions
			@EnabledIfSystemProperty(named = "java.vm.vendor", matches = "Oracle.*")
			@DisabledIfSystemProperty(named = "file.separator", matches = "[/]")
		Environment Variable Conditions
			@EnabledIfEnvironmentVariable(named = "GDMSESSION", matches = "ubuntu")
			@DisabledIfEnvironmentVariable(named = "LC_TIME", matches = ".*UTF-8.")


https://www.baeldung.com/java-system-stubs
	set os env variable
		pom.xml
			<!-- for JUnit 5 -->
			<dependency>
				<groupId>uk.org.webcompere</groupId>
				<artifactId>system-stubs-jupiter</artifactId>
				<version>1.1.0</version>
				<scope>test</scope>
			</dependency>

		xxxxTest.java
			@ExtendWith(SystemStubsExtension.class)
			class xxxxTest {
				....
				@SystemStub
				private EnvironmentVariables environmentVariables;
				// 或是 private EnvironmentVariables environmentVariables = new EnvironmentVariables("ENV", "value1");
				// 或是 private EnvironmentVariables environmentVariables = new EnvironmentVariables().set("ENV", "value1").set("ENV2", "value2");

				@Test
				void givenEnvironmentCanBeModified_whenSetEnvironment_thenItIsSet() {
					environmentVariables.set("ENV", "value1");

					assertThat(System.getenv("ENV")).isEqualTo("value1");
				}
			}
