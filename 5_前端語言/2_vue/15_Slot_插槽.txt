	CustomTitle.vue
		<template>
			<div class='custom-title'>
			  <section>
			    <Icon><component :is="icons[icon]" /></Icon>
				<p>{{title}}</p>
			  </section>

			  <section>
			    <input v-show="inputText" placeholder="inputText" />
				<button v-show="btn2Text && !hideBtn">{{btn2Text}}</button>
				<button v-show="!hideBtn">{{btnText}}</button>
			  </section>
			</div>
		</template>

		<script setup>
			import {Icon} from '@vions/utils'
			import {Document, Warning, Close} from '@vicons/carbon'

			definePros({
			  title: {type: String, defult: 'PageTile' }
			  icon: {type: String, defult: 'Document' }
			  hideBtn: {type: Boolean, defult: false }
			  inputText: {type: String, defult: '' }
			  btnText: {type: String, defult: '編輯' }
			  btn2Text: {type: String, defult: '' }
			})

			const icons = {
			  Document: Document,
			  Warning: Warning,
			  Close: Close,
			}

		</script>
