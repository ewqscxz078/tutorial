store
	count.ts
		import {defineStore} from 'pinia'

		export const useCountStore = defineStore('count',{

		  // 裡面放置的是一個一個的動作方式，用於響應元件中的動作
		  actions:{
		    increment(value){
			  // console.log('increment被叫了',value)

			  // 修改資料(this 是當前的 store)
			  // console.log(this.sum)

			  if(this.sum < 10) {
			    this.sum += value
			  }
			}
		  }

		  // 真正放資料的地方
		  state(){
		    return {
			  sum:6,
			  school:'sdfsfda',
			  address:'xxx科技園'
			}
		  }
		});

  import {createApp} from 'vue'

  import App from './App.vue'

  // 第一步 : 引入 pinia

  // 第二步 : 創建 pinia
  const pinia = createPinia()

  // 第三步 : 安裝 pinia
  app.use(pinia)

  app.mount('#App')
