setup的語法糖

		<template>
		  <div class="person">
		    <h2>姓名:{{name}}</h2>
			<h2>年齡:{{age}}</h2>
			<button @click="changeName">修改名字</button>
			<button @click="changeAge">修改年齡</button>
			<button @click="showTel">查看聯絡方式</button>
			<hr>
		  </div>
		</template>

        // 原始寫法
		<script lang="ts">
		    export default() {
			  name: 'Person',
			  setup(){

			    let name = '張三'
			    let age = 18
			    let tel = '123456789'

				let x = d
				// vue3 無法讀取到 vue2 的 data 變數

			    function changeName(){
				  name= 'zhange-san' // 注意: 這樣修改 name, 頁面是沒有變化的
				  console.log(name)  // name 確實改了, 但 name 不是響應式的
				},
				function changeAge(){
				  age += 1			// 注意: 這樣修改 age, 頁面是沒有變化的
				  console.log(age)	// age 確實改了, 但 age 不是響應式的
				},
				function showTel(){
				  alert(this.tel)
				}

				// 將資料與方法交出去，模板才可以使用
				return {name,age, changeName,changeAge,showTel}
			  }
			}
		</script>

        // 多一塊 script : for 設定控制元件的名稱，預設名稱是元件檔名
		// 如果沒裝 plugins 要改預設元件名稱時用
		<script lang="ts">
		  export default {
		    name: 'Person234'
		  }
		</script>
