hooks
	1.讓功能的 資料、方法放在一起(封裝、模組化)
	2.命名有講究如 useDog、useTeacher
	3.才發揮出 vue3 組合式寫法威力

hooks\useSum.ts
	import {ref,onMounted,computed} from 'vue'
	// npm i axios
	import axios from 'axios'

	export default function(){
	  // 資料
	  let sum = ref(0)
	  let bigSum = computed(()=>{
	     return sum.value * 10
	  }

	  // 方法
	  function add(){
		sum.value += 1
	  }

	  // 鉤子
	  onMounted(()=>{
	    add()
	  })

	  // 向外提供東西
	  return {sum,add,bigSum}
	}
