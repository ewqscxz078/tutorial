components/
	Count.vue
		<template>
			<div class="count">
			  <h2>當前求和為: {{ sum }}</h2>
			  <select v-model.number="n">
			    <option value="1">1</option>
				<option value="2">2</option>
				<option value="3">3</option>
			  </select>
			  <button @click="add">加</button>
			  <button @click="minus">減</button>

			</div>
		</template>

		<script lang='ts' setup>
		  import {ref} from "vue";

		  // data
		  let sum = ref(1) // 當前求和
		  let n = ref(1)	// 用戶選擇的數字

		  function add(){
		    sum.value += n.value
		  }
		  function minus(){
		    sum.value -= n.value
		  }
		</script>

		<style scoped>
		  .count {
		    backgroud-color: skyblue;
			padding: 10px;
			bordering: 10px;
			box-shadow: 0 0 10px;
		  }
		  select,button {
		    margin: 0 5px;
			height: 25px;
		  }
		</style>


	LoveTalk.vue

		<template>
			<div class="talk">
			  <button @click="getLoveTalk">獲取一句土味情話</button>
			  <ul>
			    <li v-for="talk in talkList" :key="talk.id">{{talk.title}}</li>
			  </ul>
			</div>
		</template>

		<script lang='ts' setup>
		  import {reactive} from "vue";
		  import axios from "axios";
		  import { nanoid } from 'nanoid' // npm i nanoid

		  // data
		  let talkList = reactive({
		    {id: 'sdf01', title: '今天你有點怪,哪裡怪?怪好看的!' },
			{id: 'sdf02', title: '草莓、藍莓、蔓越莓、你今天想我了沒' },
			{id: 'sdf03', title: '心裡給你留了一塊地，我的死心踏地' }
		  })

		  function getLoveTalk(){
			// 發請求，下面這寫法是連續解構賦值+重命名
		    let {data:{content:title}} = await axios.get('https://api.uomg.com/api.rand.qinghua?format=json')
			//console.log(result.data.content)

			// 把請求回來的字符串包裝成一個對象
			let obj = {id:nanoid(),title}
			// 放到數組中
			talkList.unshift(obj)
		  }
		</script>

		<style scoped>
		  .talk {
		    backgroud-color: orange;
			padding: 10px;
			bordering: 10px;
			box-shadow: 0 0 10px;
		  }
		  select,button {
		    margin: 0 5px;
			height: 25px;
		  }
		</style>
