store
	count.ts
		import {defineStore} from 'pinia'

		export const useCountStore = defineStore('count',{

		  // 裡面放置的是一個一個的動作方式，用於響應元件中的動作
		  actions:{
		    increment(value:number){
			  // console.log('increment被叫了',value)

			  // 修改資料(this 是當前的 store)
			  // console.log(this.sum)

			  if(this.sum < 10) {
			    this.sum += value
			  }
			}
		  }

		  // 真正放資料的地方
		  state(){
		    return {
			  sum:1,
			  school:'sdfsfda',
			  address:'xxx科技園'
			}
		  }
		});

	loveTalk.ts
		import {defineStore} from 'pinia'
		import axios from 'axios'
		import {nanoid} from 'nanoid'

		export const useTalkStore = defineStore('talk',{

		  actions:{
		    async getATalk(value){
				// 發請求，下面這寫法是連續解構賦值+重命名
				let {data:{content:title}} = await axios.get('https://api.uomg.com/api.rand.qinghua?format=json')
				//console.log(result.data.content)

				// 把請求回來的字符串包裝成一個對象
				let obj = {id:nanoid(),title}
				// 放到數組中
				talkList.unshift(obj)
			}
		  }

		  // 真正放資料的地方
		  state(){
		    return {
			  talklist:[
				{id: 'sdf01', title: '今天你有點怪,哪裡怪?怪好看的!' },
				{id: 'sdf02', title: '草莓、藍莓、蔓越莓、你今天想我了沒' },
				{id: 'sdf03', title: '心裡給你留了一塊地，我的死心踏地' }
			  ]
			}
		  }
		});

components/
	Count.vue
		<template>
			<div class="count">
			  <h2>當前求和為: {{ sum }}</h2>
			  <h3>歡迎來到: {{ school }}, 坐落於{{ address }}</h3>
			  <select v-model.number="n">
			    <option value="1">1</option>
				<option value="2">2</option>
				<option value="3">3</option>
			  </select>
			  <button @click="add">加</button>
			  <button @click="minus">減</button>

			</div>
		</template>

		<script lang='ts' setup>
		  import {ref,reactive} from "vue";
		  import {useCountStore} from '@/store/count'
		  import {storeToRefs} from 'pinia'

		  const countStore = useCountStore()
		  // const {sum,school,address} = countStore 拿回來的時候不是響應式
		  // const {sum,school,address} = toRefs(countStore) // 代價是所有東西都變成響應式，但我不需要所有

		  // storeToRefs 只會關注 store 中的資料，不會對方法進行 refs 轉換包裝
		  const {sum,school,address} = storeToRefs(countStore)

		  let n = ref(1)	// 用戶選擇的數字

		  function add(){
		    countStore.increment(n.value)
		  }
		  function minus(){
		    countStore.sum -= n.value
		  }
		</script>

main.ts
  import {createApp} from 'vue'

  import App from './App.vue'

  // 第一步 : 引入 pinia
  import {createPinia} from 'pinia'

  const app = createApp(App))

  // 第二步 : 創建 pinia
  const pinia = createPinia()

  // 第三步 : 安裝 pinia
  app.use(pinia)

  app.mount('#App')
